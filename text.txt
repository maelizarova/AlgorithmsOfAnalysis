def calculate_stat_score(data: pd.DataFrame):
    """ 
    Вспомогательная функция для расчета статистик по скорам (count, mean, max, min, percentiles)
    Input:
        data - датафрейм со скорами
    Output:
        таблица со статистиками по скорам
    
    """ 
    base_stat = data['score'].agg(['count', 'mean', 'max', 'min']).reset_index()
    base_stat.columns =  ['metric_name','metric_value']

    q_dic ={}
    for q, label in zip([0.1, 0.25, 0.50, 0.75, 0.9], ['p_10', 'p_25', 'p_50', 'p_75', 'p_90']):
        q_dic[label] = data['score'].quantile(q)
    q_stat = pd.DataFrame.from_dict(q_dic, orient = 'index').reset_index()
    q_stat.columns = ['metric_name' , 'metric_value']

    return pd.concat( [base_stat, q_stat], ignore_index =True)

def calculate_na_feat(data: pd.DataFrame, features: list):
    """ 
    Вспомогательная функция для расчета пропусков по фичам 
    Input:
        data - датафрейм с фичами
        features - список фичей
    Output:
        таблица с пропусками
    
    """ 
    cat_cols = data[features].select_dtypes('object').columns.tolist()
    num_cols = data[features].select_dtypes('number').columns.tolist()

    if num_cols:
        df_na_num = pd.DataFrame(data[num_cols].isnull().agg(['sum', 'mean'])).T.reset_index()
        df_na_num.columns = ['feature_name', 'nan_count', 'nan_share']
        df_na_num = df_na_num.melt(id_vars = 'feature_name', var_name = 'metric_name' , value_name = 'metric_value')
        df_na_num['feature_type'] = 'numeric'
    else:
        df_na_num = pd.DataFrame()
    
    if cat_cols:
        df_na_cat = pd.DataFrame(data[cat_cols].isnull().agg(['sum', 'mean'])).T.reset_index()
        df_na_cat.columns = ['feature_name', 'nan_count', 'nan_share']
        df_na_cat = df_na_cat.melt(id_vars = 'feature_name', var_name = 'metric_name' , value_name = 'metric_value')
        df_na_cat['feature_type'] = 'categorial'
    else:
        df_na_cat = pd.DataFrame()
    
    return pd.concat( [df_na_num, df_na_cat], ignore_index =True) 


def calculate_metric_by_window(target_dic: dict):
    """ 
    Вспомогательная функция для расчета метрик (roc_auc, gini) по разным вызреваниям таргета
    Input:
        target_dic - словарь с таргетами
    Output:
        таблица с метриками
    
    """ 
    metric_dic = {}
    for k, v in target_dic.items():
        if not v.empty:
            v = v[v.target.notna()]
            metric_dic[f"TARGET_COUNT@{k}"] = v['target'].sum()
            metric_dic[f"TARGET_RATE@{k}"] = v['target'].mean()
            if len(np.unique(v['target']))==2:
                metric_dic[f"ROC_AUC@{k}"] = roc_auc_score(v['target'], v['score'])
            else:
                metric_dic[f"ROC_AUC@{k}"] = np.nan
            
            metric_dic[f"GINI@{k}"] = 2 * metric_dic[f"ROC_AUC@{k}"] - 1
         
    return pd.DataFrame.from_dict(metric_dic, orient = 'index', columns = ['metric_value']).reset_index().rename(columns = {'index': 'metric_name'})


def calculate_target_share_by_window(target_dic: dict):
    """ 
    Вспомогательная функция для расчета доль таргета по бинам скора для разных вызреваний таргета
    Input:
        target_dic - словарь с таргетами
    Output:
        таблица с долями таргета по бинам скора для разных вызреваний таргета
    
    """ 
    data_list = []
    for k, v in target_dic.items():
        if not v.empty:
            v = v[v.target.notna()].copy()
            v['ranked_score'] = v['score'].rank(method = 'first')
            v['decile'] = pd.qcut(v['ranked_score'], 10, labels=range(1,11))
            
            df_group = v.groupby('decile', observed = False)[['target']].sum()
            df_group['target_share'] = df_group['target'].apply(lambda x: 0 if df_group['target'].sum() == 0 else x/df_group['target'].sum())
            
            mapping = { 10 : '1-10', 9 :'11-20', 8 : '21-30', 7 : '31-40', 6 : '41-50', 
                                5 : '51-60', 4 : '61-70', 3 : '71-80', 2 : '81-90', 1 : '91-100'}
            
            df_group.index = df_group.index.map(mapping)
            df_group.sort_values(by = 'decile', ascending = False, inplace = True)
            df_group.columns = [f'TARGET_COUNT@{k}', f'TARGET_SHARE@{k}']
            df_group = df_group.reset_index()
            
            data_list.append(df_group.melt(id_vars = 'decile', var_name = 'metric_name' , value_name = 'metric_value'))

    return pd.concat(data_list, axis =0)

def monitor_binary_clf_score(config, execution_date):
    """ 
    Функция обновления psi и статистик по скорам (count, mean, max, min, percentiles)
    Input:
        config - словарь с параметрами модели
        execution_date - дата расчета (логическя дата, дата выполнения дага) не объязательно равняется текущей дате
    Output:
        None    
    """ 
    logger = get_run_logger()
    logger.info(f"Updating score PSI for {config['model_name']}")
    
    # скрипты, чтобы вытащить скоры, если необходимы какие-то манипуляции (для онлайн моделей)
    if config.get('score_script_path'):
        with open(config['score_script_path'], encoding = 'utf-8') as f:
            query = f.read()
        try:
            with engine.begin() as conn:
                conn.execute(text(query.format(execution_date=execution_date)))
                
        except SQLAlchemyError as e:
            logger.warning(
                f"Score collection for {config['model_name']} failed on date {execution_date}."
                f"Error: {e}"
            )
            return None  
            
    current_data = pd.read_sql(f"""select /*+ PARALLEL(8) */ * from {config['source_score']} """,  con=engine)       
    base_data = pd.read_sql(f"""select /*+ PARALLEL(8) */ * from {config['benchmark_score']} """,  con=engine)
    
    if current_data.empty:
        logger.warning(f"No data for {config['model_name']} scores on {execution_date}")
        return None

    # расчет psi 
    pa = PSIAnalyzer(base_data, current_data , feature = 'score')
    psi_value = pa.analyze() 
    
    df_psi = pd.DataFrame(columns=['model_name', 'report_date', 'psi'])
    df_psi.loc[0] = [config['model_name'], execution_date,  psi_value]
    df_psi['report_date'] = pd.to_datetime(df_psi['report_date'], format ='%Y-%m-%d')

    # кастомный класс для записи в Oracle 
    writer_1 = SQLWriter(
        df_psi, 
        'nao_model_monitor_clf_score_psi_table',
        {
            "model_name": types.VARCHAR(100),
            "report_date": types.Date(),
            "psi": types.Float()
        },
        config['model_name'], 
        execution_date
    )
    writer_1.write_data(engine)

    logger.info(f"Updating score stats for {config['model_name']}")

    week_flg = config.get('week_flg', False)
    # расчет статистик по скорам
    # для онлайн моделей считаем на 1 день и неделю
    if config['is_online'] or week_flg:
        df_stat_7d = calculate_stat_score(current_data)
        df_stat_7d['metric_name'] = df_stat_7d['metric_name'] + '_7d'
        df_stat_1d = calculate_stat_score(current_data[current_data[config['date_col']] == current_data[config['date_col']].max()])
        df_stat_1d['metric_name'] = df_stat_1d['metric_name'] + '_1d'
        df_stat = pd.concat([df_stat_7d, df_stat_1d], ignore_index =True)
    else:    
        df_stat = calculate_stat_score(current_data)
        df_stat['metric_name'] = df_stat['metric_name'] + '_1d'

    df_stat['model_name'] = config['model_name']
    df_stat['report_date'] =  execution_date
    df_stat['report_date'] = pd.to_datetime(df_stat['report_date'], format = "%Y-%m-%d")
    df_stat = df_stat[['model_name', 'report_date','metric_name', 'metric_value']]
    
    # кастомный класс для записи в Oracle 
    writer_2 = SQLWriter(
        df_stat, 
        'nao_model_monitor_clf_score_stat_table',
        {
            "model_name": types.VARCHAR(100),
            "report_date": types.Date(),
            "metric_name": types.VARCHAR(100),
            "metric_value": types.Float()
        },
        config['model_name'], 
        execution_date
    )
    writer_2.write_data(engine)
    logger.info(f"Finished score PSI and stats for {config['model_name']}")
