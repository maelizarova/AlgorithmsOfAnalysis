import plotly.express as px

pt = (
    sample[sample["year_month"] < "2024-11-01"]
    .pivot_table(
        values=["client_did"],
        index=["year_month"],
        aggfunc="count",
        margins=False,
    )
    .reset_index()
)

fig_sample = px.bar(
    pt,
    y="year_month",
    x="client_did",
    orientation="h",
    title="Количество клиентов по месяцам",
)

# как в pandas barh: месяцы сверху вниз по возрастанию
fig_sample.update_layout(
    yaxis=dict(type="category", categoryorder="category ascending"),
    legend_title_text="",
)

fig_sample.show()  # в ноутбуке — опционально
