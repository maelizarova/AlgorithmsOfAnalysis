-- 1) день + ROWID
CREATE TABLE tmp_scores_day AS
SELECT d.ROWID AS row_id, d.*
FROM   vmr_cc_active_base_scores d
WHERE  d.report_date = DATE '2026-05-10';

-- 2) какие row_id удалять (дубли, rn >= 2)
CREATE TABLE tmp_scores_del AS
SELECT row_id
FROM (
  SELECT row_id,
         ROW_NUMBER() OVER (
           PARTITION BY client_did, report_date, score, decile, lvl1, lvl2
           ORDER BY row_id
         ) AS rn
  FROM   tmp_scores_day
)
WHERE rn >= 2;

-- 3) удаление чанками из основной
DECLARE
  CURSOR c_ids IS
    SELECT row_id FROM tmp_scores_del;

  TYPE t_ids IS TABLE OF ROWID;
  l_ids t_ids;

  c_limit PLS_INTEGER := 50000;
  l_total PLS_INTEGER := 0;
BEGIN
  OPEN c_ids;
  LOOP
    FETCH c_ids BULK COLLECT INTO l_ids LIMIT c_limit;
    EXIT WHEN l_ids.COUNT = 0;

    FORALL i IN 1 .. l_ids.COUNT
      DELETE FROM vmr_cc_active_base_scores
      WHERE  ROWID = l_ids(i);

    l_total := l_total + l_ids.COUNT;
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('chunk=' || l_ids.COUNT || ' total=' || l_total);
  END LOOP;
  CLOSE c_ids;
  DBMS_OUTPUT.PUT_LINE('DONE total=' || l_total);
END;
/
