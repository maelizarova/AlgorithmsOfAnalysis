# === Фильтр фичей по PSI (исправленный расчёт тренда) ===
# Используйте эту ячейку вместо предыдущей; в конце перезаписывается to_del_all.
#
# Что было не так в linregress(range(len), col)) по всему ряду:
# 1) Порядок периодов — без sort_index() тренд мог считаться не по времени.
# 2) Синтетический референс (2000-01-01, пул базовых month) с PSI≈0 тянет slope вверх.
# 3) NaN в PSI не отбрасывались — регрессия могла сломаться.
# 4) PSI_TREND=0.05 — это порог max(PSI) для ветки «тренд», не порог наклона (как в шаблоне).

_psi_sorted = _psi_for_filter.sort_index()

# период-референс toolbox не участвует в тренде (только в max по всему ряду)
_psi_ref_label = PSI_REF_PERIOD if "PSI_REF_PERIOD" in dir() else _psi_sorted.index[0]

to_del_fixed = []
trend_only = []   # ушли только по тренду, max < PSI_THRESHOLD

for col in _psi_sorted.columns:
    full = _psi_sorted[col]
    max_psi = full.max()

    # фильтр 1: любой период выше жёсткого порога
    if max_psi >= PSI_THRESHOLD:
        to_del_fixed.append(col)
        continue

    # фильтр 2: значимый рост PSI по month после референса
    for_trend = full.drop(index=_psi_ref_label, errors="ignore").dropna()
    if len(for_trend) < 3:
        continue

    lrgs = linregress(np.arange(len(for_trend)), for_trend.to_numpy())
    trend_up = (lrgs.slope > 0) and (lrgs.pvalue < 0.05) and (max_psi > PSI_TREND)
    if trend_up:
        to_del_fixed.append(col)
        trend_only.append(col)

to_del_all = to_del_fixed

print(f"Удаляем фичей: {len(to_del_all)} (остаётся {len(features) - len(to_del_all)})")
print(f"  из них только по тренду (max < {PSI_THRESHOLD}): {len(trend_only)}")
print(f"  по max >= {PSI_THRESHOLD}: {len(to_del_all) - len(trend_only)}")

# сравнение со старой формулой (без исправлений)
to_del_legacy = []
for col in _psi_for_filter.columns:
    if _psi_for_filter[col].max() >= PSI_THRESHOLD:
        to_del_legacy.append(col)
        continue
    lrgs = linregress(range(len(_psi_for_filter[col])), _psi_for_filter[col])
    if ((lrgs.slope > 0) & (lrgs.pvalue < 0.05) & (_psi_for_filter[col].max() > PSI_TREND)):
        to_del_legacy.append(col)

only_legacy = set(to_del_legacy) - set(to_del_all)
only_fixed = set(to_del_all) - set(to_del_legacy)
print(f"\nСравнение со старой ячейкой:")
print(f"  только старая удалила бы: {len(only_legacy)}")
print(f"  только новая удаляет:      {len(only_fixed)}")
if only_legacy:
    print("  примеры (лишнее удаление старой):", list(only_legacy)[:5])
if only_fixed:
    print("  примеры (новая ловит тренд):", list(only_fixed)[:5])
