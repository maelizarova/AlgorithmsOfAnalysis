import numpy as np
import plotly.graph_objects as go
from plotly.subplots import make_subplots


def target_distribution(
    df,
    target_cols,
    ncols=2,
    log_scale=False,
    bins=30,
    figsize=None,
    show=True,
):
    """
    Гистограммы и boxplot по списку таргетов.
    log_scale — log1p по значениям для гистограммы.
    bins — число столбиков (мало данных в JSON → без 413 в ClearML).
    """
    if isinstance(target_cols, str):
        target_cols = [target_cols]

    n = len(target_cols)
    nrows = (n + ncols - 1) // ncols
    ncols_plot = ncols * 2

    if figsize is None:
        figsize = (5 * ncols_plot, max(5, 4 * nrows))

    width_px = int(figsize[0] * 80)
    height_px = int(figsize[1] * 80)

    fig = make_subplots(
        rows=nrows,
        cols=ncols_plot,
        subplot_titles=[
            t for col in target_cols for t in (col, f"{col} (box)")
        ],
        horizontal_spacing=0.08,
        vertical_spacing=0.15 if nrows > 1 else 0.12,
    )

    for i, col in enumerate(target_cols):
        r = i // ncols + 1
        c_hist = (i % ncols) * 2 + 1
        c_box = c_hist + 1

        x = df[col].dropna().astype(float)

        x_hist = x
        if log_scale and x_hist.max() > 0:
            x_hist = np.log1p(x_hist)

        # гистограмма: только бины, не весь столбец
        counts, edges = np.histogram(x_hist, bins=bins)
        centers = (edges[:-1] + edges[1:]) / 2
        widths = np.diff(edges)

        fig.add_trace(
            go.Bar(
                x=centers,
                y=counts,
                width=widths,
                marker=dict(line=dict(color="black", width=0.5)),
                opacity=0.7,
                showlegend=False,
                hovertemplate="значение: %{x:.4g}<br>число: %{y}<extra></extra>",
            ),
            row=r,
            col=c_hist,
        )

        fig.add_trace(
            go.Box(
                y=x,
                showlegend=False,
                hovertemplate="%{y:.4g}<extra></extra>",
            ),
            row=r,
            col=c_box,
        )

    fig.update_layout(
        width=width_px,
        height=height_px,
        autosize=False,
        title_text="Распределение таргетов",
        margin=dict(l=60, r=40, t=80, b=60),
        bargap=0,
    )

    if show:
        fig.show()

    return fig
