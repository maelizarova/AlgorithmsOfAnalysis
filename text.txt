DELETE FROM your_schema.your_table
WHERE ROWID IN (
  SELECT rid
  FROM (
    SELECT ROWID AS rid,
           ROW_NUMBER() OVER (
             PARTITION BY col1, col2, col3, ...  -- все колонки, кроме служебных
             ORDER BY ROWID
           ) AS rn
    FROM your_schema.your_table
  )
  WHERE rn > 1
);
COMMIT;
