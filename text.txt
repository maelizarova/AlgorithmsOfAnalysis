CREATE TABLE ema_sample_es19682_target0 AS
WITH months AS (
  SELECT ADD_MONTHS(DATE '2025-07-01', LEVEL - 1) AS month_start
  FROM dual
  CONNECT BY LEVEL <= 9
)
SELECT /*+ PARALLEL(4) */
  client_did,
  month_start,
  report_date
FROM (
  SELECT
    ac.client_did,
    ac.month_start,
    ac.report_date,
    ROW_NUMBER() OVER (
      PARTITION BY ac.month_start
      ORDER BY DBMS_RANDOM.VALUE
    ) AS rn_month
  FROM (
    SELECT
      client_did,
      month_start,
      report_date
    FROM (
      SELECT
        c.client_did,
        m.month_start,
        TRUNC(m.month_start)
          + FLOOR(
              DBMS_RANDOM.VALUE(
                0,
                (LAST_DAY(m.month_start) - m.month_start) + 1
              )
            ) AS report_date,
        ROW_NUMBER() OVER (
          PARTITION BY c.client_did, m.month_start
          ORDER BY DBMS_RANDOM.VALUE
        ) AS rn_client_month
      FROM DM_MRK.DM_MRK_contract c
      JOIN months m
        ON c.open_dt <= LAST_DAY(m.month_start)
       AND (c.fact_close_dt IS NULL OR c.fact_close_dt >= m.month_start)
      WHERE c.open_dt <= LAST_DAY(ADD_MONTHS(DATE '2025-07-01', 8))
        AND (c.fact_close_dt IS NULL OR c.fact_close_dt >= DATE '2025-07-01')
    )
    WHERE rn_client_month = 1
  ) ac
) q
WHERE rn_month <= 66000;
