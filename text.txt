from prefect import flow, task, Task, get_run_logger
from prefect.task_runners import ConcurrentTaskRunner
from models.binary_classification import BinaryClassificationMonitor
from prefect.blocks.notifications import CustomWebhookNotificationBlock
from toolbox.prefect.notification import make_email_flow_notifier
from datetime import datetime
import json
import glob
import os

config_paths = glob.glob(os.path.join('configs', '*.json'))
execution_date = datetime.now().date().strftime('%Y-%m-%d')
MAX_WORKERS = 4
RETRIES = 1
RETRY_DELAY_SECONDS = 60

custom_webhook_block = CustomWebhookNotificationBlock.load("mail-to-analytics") 
on_flow_failure = make_email_flow_notifier(custom_webhook_block)


def run_single_model(config_path, execution_date):
    with open(config_path) as f:
            config = json.load(f)        
    logger = get_run_logger()
    logger.info(f"Running monitor for {config['model_name']}")
    monitor = BinaryClassificationMonitor(config, execution_date)
    monitor.run(RETRIES, RETRY_DELAY_SECONDS)
    logger.info(f"Finished monitor for {config['model_name']}")
    

 
@flow(name="bianry_clf_monitor", retries=RETRIES, retry_delay_seconds=RETRY_DELAY_SECONDS, task_runner=ConcurrentTaskRunner(max_workers=MAX_WORKERS), on_failure =[make_email_flow_notifier])
def run_all_models():    
    task_runs = []

    for config_path in config_paths:
        with open(config_path) as f:
            config = json.load(f)
        model_name = config['model_name']
        t = Task(fn=run_single_model, 
                   name = f"run_model_{model_name}", 
                   task_run_name = f"run_model[{model_name}]",
                   retries=RETRIES,
                   retry_delay_seconds=RETRY_DELAY_SECONDS)
        fut = t.submit(config_path, execution_date)
        task_runs.append(fut) 
        
    _ = [t.result() for t in task_runs]


if __name__ == "__main__":
    run_all_models()
