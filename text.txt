CREATE TABLE YOUR_SCHEMA.T_PREMIUM_IDS AS
SELECT DISTINCT CLIENT_UID
FROM YOUR_SCHEMA.YOUR_TABLE
WHERE REPORT_DT >= DATE '2025-07-01'
  AND IS_ACTIVE_PACK_PREMIUM_FLG = 1;

CREATE TABLE YOUR_SCHEMA.T_PREMIUM_ROWS AS
SELECT s.*
FROM YOUR_SCHEMA.YOUR_TABLE s
JOIN YOUR_SCHEMA.T_PREMIUM_IDS i ON i.CLIENT_UID = s.CLIENT_UID
WHERE s.REPORT_DT >= DATE '2025-07-01';

CREATE TABLE YOUR_SCHEMA.T_PREMIUM_TARGET AS
SELECT *
FROM (
  SELECT
    r.*,
    LAG(IS_ACTIVE_PACK_PREMIUM_FLG) OVER (
      PARTITION BY CLIENT_UID
      ORDER BY REPORT_DT, CLIENT_IBSO_ID
    ) AS PREV_PREMIUM_FLG
  FROM YOUR_SCHEMA.T_PREMIUM_ROWS r
) q
WHERE q.PREV_PREMIUM_FLG = 0
  AND q.IS_ACTIVE_PACK_PREMIUM_FLG = 1
  AND q.PORTF_RUB_AMT > 2500000;
