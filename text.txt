import pandas as pd


def decile_target(
    df: pd.DataFrame,
    score_col: str,
    target_col: str,
    id_col: str,
    q: int = 10,
    as_percent: bool = True,
    current_debt_amt_col: str = "current_debt_amt",
    current_debt_amt_3m_col: str = "current_debt_amt_3m",
    old_limit_col: str = "old_limit",
    new_limit_col: str = "new_limit",
) -> pd.DataFrame:
    """
    Анализ таргета по децилям скора.

    Параметры:
    ----------
    df : pd.DataFrame
        Исходные данные
    score_col : str
        Название колонки со скором/вероятностью
    target_col : str
        Название колонки с таргетом (0/1)
    id_col : str
        Название колонки с ID клиента
    q : int, optional
        Количество квантилей (по умолчанию 10)
    as_percent : bool, optional
        Возвращать ratio в процентах (по умолчанию True)
    current_debt_amt_col : str, optional
        Название колонки с текущей задолженностью
    current_debt_amt_3m_col : str, optional
        Название колонки с задолженностью через 3 месяца
    old_limit_col : str, optional
        Название колонки с лимитом до клипа
    new_limit_col : str, optional
        Название колонки с лимитом после клипа

    Возвращает:
    -----------
    pd.DataFrame
        DataFrame с колонками: decile, client_all, target_0, target_1, ratio,
        avg_debt_growth, avg_old_limit, avg_new_limit
    """
    metric_cols = [
        current_debt_amt_col,
        current_debt_amt_3m_col,
        old_limit_col,
        new_limit_col,
    ]
    base_cols = [score_col, target_col] + ([id_col] if id_col else [])
    d = df[base_cols + metric_cols].dropna(subset=base_cols).copy()

    # Дециль: 1 = максимум score
    d["decile"] = pd.qcut(-d[score_col], q=q, labels=False, duplicates="drop") + 1
    d["debt_growth"] = d[current_debt_amt_3m_col] - d[current_debt_amt_col]

    # target_0 / target_1 по децилям
    pivot = (
        d.groupby(["decile", target_col])[id_col]
        .nunique()
        .unstack(fill_value=0)
        .rename(columns={0: "target_0", 1: "target_1"})
        .reset_index()
        .sort_values("decile")
    )

    # всего клиентов в дециле
    pivot["client_all"] = pivot["target_0"] + pivot["target_1"]

    # доля target=1 в дециле от общего числа target=1 по всей выборке
    total_pos = pivot["target_1"].sum()
    pivot["ratio"] = pivot["target_1"] / total_pos if total_pos > 0 else 0.0

    # средние показатели по децилям
    metrics = (
        d.groupby("decile")
        .agg(
            avg_debt_growth=("debt_growth", "mean"),
            avg_old_limit=(old_limit_col, "mean"),
            avg_new_limit=(new_limit_col, "mean"),
        )
        .reset_index()
    )
    pivot = pivot.merge(metrics, on="decile", how="left")

    # формат процентов по желанию
    if as_percent:
        pivot["ratio"] = (pivot["ratio"] * 100).round(2).astype(str) + "%"

    # привести тип decile к int (а не category)
    pivot["decile"] = pivot["decile"].astype(int)

    # финальные колонки в нужном порядке
    pivot = pivot[
        [
            "decile",
            "client_all",
            "target_0",
            "target_1",
            "ratio",
            "avg_debt_growth",
            "avg_old_limit",
            "avg_new_limit",
        ]
    ]
    pivot.columns.name = None
    return pivot
