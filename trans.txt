from __future__ import annotations

from pathlib import PurePosixPath
from uuid import UUID, uuid4

import yaml
from fastapi import HTTPException, status

from ..config import PathsSettings
from ..exceptions import StorageError
from ..storage import StorageClient, StorageObject
from ..utils import parse_s3_uri


class AdminService:
    PROFILES_FILE_NAME = "profiles.yaml"

    def __init__(self, storage: StorageClient, paths: PathsSettings):
        self.storage = storage
        self.paths = paths

    @staticmethod
    def _safe_decode(payload: bytes) -> str:
        return payload.decode("utf-8", errors="replace")

    @staticmethod
    def _join_uri(prefix: str, file_name: str) -> str:
        return f"{prefix.rstrip('/')}/{file_name.lstrip('/')}"

    @staticmethod
    def _is_uuid(value: str) -> bool:
        try:
            UUID(value)
            return True
        except ValueError:
            return False

    @staticmethod
    def _validate_uuid(value: str, field_name: str) -> str:
        try:
            return str(UUID(value))
        except ValueError as exc:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail=f"'{field_name}' должен быть корректным UUID.",
            ) from exc

    @staticmethod
    def _validate_yaml(raw_yaml: str) -> None:
        if not raw_yaml.strip():
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="YAML-тело запроса пустое.",
            )
        try:
            yaml.safe_load(raw_yaml)
        except yaml.YAMLError as exc:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail=f"Некорректный YAML: {exc}",
            ) from exc

    @staticmethod
    def _normalize_case_id(case_id: str) -> str:
        normalized = case_id.strip()
        if not normalized:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="'case_id' не может быть пустым.",
            )
        if "/" in normalized or "\\" in normalized or normalized in {".", ".."}:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="'case_id' должен быть именем файла без путей.",
            )

        suffix = PurePosixPath(normalized).suffix.lower()
        if suffix == "":
            return normalized
        if suffix in {".yaml", ".yml"}:
            case_id_without_ext = PurePosixPath(normalized).stem.strip()
            if case_id_without_ext:
                return case_id_without_ext
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="'case_id' не может быть пустым.",
            )

        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="'case_id' должен быть без расширения или иметь .yaml/.yml.",
        )

    @staticmethod
    def _normalize_session_log_id(session_log_id: str) -> str:
        normalized = session_log_id.strip()
        if not normalized:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="'session_log_id' не может быть пустым.",
            )
        if "/" in normalized or "\\" in normalized or normalized in {".", ".."}:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="'session_log_id' должен быть именем файла без путей.",
            )

        suffix = PurePosixPath(normalized).suffix.lower()
        if suffix == "":
            return normalized
        if suffix == ".json":
            log_id_without_ext = PurePosixPath(normalized).stem.strip()
            if log_id_without_ext:
                return log_id_without_ext
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="'session_log_id' не может быть пустым.",
            )

        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="'session_log_id' должен быть без расширения или иметь .json.",
        )

    @staticmethod
    def _case_file_name(case_id: str) -> str:
        return f"{case_id}.yaml"

    @staticmethod
    def _session_log_file_name(session_log_id: str) -> str:
        return f"{session_log_id}.json"

    @staticmethod
    def _as_iso_datetime(storage_obj: StorageObject) -> str | None:
        if storage_obj.last_modified is None:
            return None
        return storage_obj.last_modified.isoformat()

    def _storage_call(self, fn, *args, **kwargs):
        try:
            return fn(*args, **kwargs)
        except StorageError as exc:
            raise HTTPException(
                status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
                detail=f"Ошибка хранилища: {exc}",
            ) from exc

    def _relative_key(self, full_key: str, uri_prefix: str) -> str | None:
        _, prefix_key = parse_s3_uri(uri_prefix)
        normalized_prefix = prefix_key.rstrip("/") + "/"
        if not full_key.startswith(normalized_prefix):
            return None
        return full_key[len(normalized_prefix) :]

    def _list_yaml_entities(self, uri_prefix: str, *, require_uuid: bool) -> list[dict]:
        objects = self._storage_call(self.storage.list, uri_prefix)
        result: list[dict] = []
        for obj in objects:
            relative_key = self._relative_key(obj.key, uri_prefix)
            if not relative_key or "/" in relative_key:
                continue
            if not relative_key.endswith((".yaml", ".yml")):
                continue
            entity_id = PurePosixPath(relative_key).stem
            if require_uuid and not self._is_uuid(entity_id):
                continue
            result.append(
                {
                    "id": entity_id,
                    "file_name": relative_key,
                    "size": obj.size,
                    "last_modified": self._as_iso_datetime(obj),
                }
            )
        result.sort(key=lambda item: item["id"])
        return result

    def _find_existing_case_uri(self, case_id: str) -> tuple[str, str] | None:
        for ext in (".yaml", ".yml"):
            file_name = f"{case_id}{ext}"
            uri = self._join_uri(self.paths.cases, file_name)
            if self._storage_call(self.storage.exists, uri):
                return file_name, uri
        return None

    def _create_yaml_entity(self, uri_prefix: str, raw_yaml: str) -> dict:
        self._validate_yaml(raw_yaml)
        entity_id = str(uuid4())
        uri = self._join_uri(uri_prefix, f"{entity_id}.yaml")
        self._storage_call(self.storage.put, uri, raw_yaml.encode("utf-8"))
        return {"id": entity_id, "uri": uri}

    def _get_yaml_entity(self, uri_prefix: str, entity_id: str, entity_name: str) -> dict:
        valid_id = self._validate_uuid(entity_id, f"{entity_name}_id")
        uri = self._join_uri(uri_prefix, f"{valid_id}.yaml")
        exists = self._storage_call(self.storage.exists, uri)
        if not exists:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail=f"{entity_name.capitalize()} не найден.",
            )
        payload = self._storage_call(self.storage.get, uri)
        return {"id": valid_id, "yaml": self._safe_decode(payload)}

    def _update_yaml_entity(
        self, uri_prefix: str, entity_id: str, raw_yaml: str, entity_name: str
    ) -> dict:
        valid_id = self._validate_uuid(entity_id, f"{entity_name}_id")
        self._validate_yaml(raw_yaml)
        uri = self._join_uri(uri_prefix, f"{valid_id}.yaml")
        exists = self._storage_call(self.storage.exists, uri)
        if not exists:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail=f"{entity_name.capitalize()} не найден.",
            )
        self._storage_call(self.storage.put, uri, raw_yaml.encode("utf-8"))
        return {"id": valid_id, "uri": uri}

    def _delete_yaml_entity(self, uri_prefix: str, entity_id: str, entity_name: str) -> dict:
        valid_id = self._validate_uuid(entity_id, f"{entity_name}_id")
        uri = self._join_uri(uri_prefix, f"{valid_id}.yaml")
        exists = self._storage_call(self.storage.exists, uri)
        if not exists:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail=f"{entity_name.capitalize()} не найден.",
            )
        self._storage_call(self.storage.delete, uri)
        return {"id": valid_id, "deleted": True}

    def list_cases(self) -> list[dict]:
        return self._list_yaml_entities(self.paths.cases, require_uuid=False)

    def create_case(self, raw_yaml: str, case_id: str | None = None) -> dict:
        self._validate_yaml(raw_yaml)
        normalized_case_id = (
            self._normalize_case_id(case_id) if case_id is not None else f"case_{uuid4().hex}"
        )
        file_name = self._case_file_name(normalized_case_id)
        uri = self._join_uri(self.paths.cases, file_name)
        if self._storage_call(self.storage.exists, uri):
            raise HTTPException(
                status_code=status.HTTP_409_CONFLICT,
                detail="Кейс с таким именем уже существует.",
            )
        self._storage_call(self.storage.put, uri, raw_yaml.encode("utf-8"))
        return {"id": normalized_case_id, "file_name": file_name, "uri": uri}

    def get_case(self, case_id: str) -> dict:
        normalized_case_id = self._normalize_case_id(case_id)
        case_uri = self._find_existing_case_uri(normalized_case_id)
        if case_uri is None:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail="Кейс не найден.",
            )
        file_name, uri = case_uri
        payload = self._storage_call(self.storage.get, uri)
        return {
            "id": normalized_case_id,
            "file_name": file_name,
            "yaml": self._safe_decode(payload),
        }

    def update_case(self, case_id: str, raw_yaml: str) -> dict:
        normalized_case_id = self._normalize_case_id(case_id)
        self._validate_yaml(raw_yaml)
        case_uri = self._find_existing_case_uri(normalized_case_id)
        if case_uri is None:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail="Кейс не найден.",
            )
        file_name, uri = case_uri
        self._storage_call(self.storage.put, uri, raw_yaml.encode("utf-8"))
        return {"id": normalized_case_id, "file_name": file_name, "uri": uri}

    def delete_case(self, case_id: str) -> dict:
        normalized_case_id = self._normalize_case_id(case_id)
        case_uri = self._find_existing_case_uri(normalized_case_id)
        if case_uri is None:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail="Кейс не найден.",
            )
        file_name, uri = case_uri
        self._storage_call(self.storage.delete, uri)
        return {"id": normalized_case_id, "file_name": file_name, "deleted": True}

    def _profiles_uri(self) -> str:
        return self._join_uri(self.paths.profiles, self.PROFILES_FILE_NAME)

    def get_profiles(self) -> dict:
        uri = self._profiles_uri()
        exists = self._storage_call(self.storage.exists, uri)
        if not exists:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail="Файл профилей не найден.",
            )
        payload = self._storage_call(self.storage.get, uri)
        return {
            "file_name": self.PROFILES_FILE_NAME,
            "uri": uri,
            "yaml": self._safe_decode(payload),
        }

    def update_profiles(self, raw_yaml: str) -> dict:
        self._validate_yaml(raw_yaml)
        uri = self._profiles_uri()
        existed = self._storage_call(self.storage.exists, uri)
        self._storage_call(self.storage.put, uri, raw_yaml.encode("utf-8"))
        return {
            "file_name": self.PROFILES_FILE_NAME,
            "uri": uri,
            "updated": True,
            "created": not existed,
        }

    def list_sessions(self) -> list[dict]:
        objects = self._storage_call(self.storage.list, self.paths.sessions)
        result: list[dict] = []
        for obj in objects:
            relative_key = self._relative_key(obj.key, self.paths.sessions)
            if not relative_key or "/" in relative_key:
                continue
            file_name = PurePosixPath(relative_key).name
            if not file_name.endswith(".json"):
                continue
            session_id = PurePosixPath(file_name).stem
            if not session_id:
                continue

            result.append(
                {
                    "id": session_id,
                    "file_name": relative_key,
                    "size": obj.size,
                    "last_modified": self._as_iso_datetime(obj),
                    "uri": obj.uri,
                }
            )
        result.sort(
            key=lambda item: (item["last_modified"] is not None, item["last_modified"] or ""),
            reverse=True,
        )
        return result

    def get_session_log(self, session_log_id: str) -> dict:
        normalized_session_log_id = self._normalize_session_log_id(session_log_id)
        file_name = self._session_log_file_name(normalized_session_log_id)
        uri = self._join_uri(self.paths.sessions, file_name)
        exists = self._storage_call(self.storage.exists, uri)
        if not exists:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail="Лог сессии не найден.",
            )
        payload = self._storage_call(self.storage.get, uri)
        return {
            "id": normalized_session_log_id,
            "file_name": file_name,
            "uri": uri,
            "content": self._safe_decode(payload),
        }

    def get_score(self, score_id: str) -> dict:
        if not score_id.strip():
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="'score_id' не может быть пустым.",
            )

        candidates = [score_id]
        if "." not in score_id:
            candidates.extend(f"{score_id}{ext}" for ext in (".json", ".yaml", ".yml", ".txt"))

        for file_name in candidates:
            uri = self._join_uri(self.paths.scores, file_name)
            if self._storage_call(self.storage.exists, uri):
                payload = self._storage_call(self.storage.get, uri)
                return {
                    "id": score_id,
                    "file_name": file_name,
                    "uri": uri,
                    "content": self._safe_decode(payload),
                }

        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="Оценка не найдена.",
        )
