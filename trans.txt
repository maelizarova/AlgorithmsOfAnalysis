# Полностью удаляем width/height из layout, если они были заданы ранее
        try:
            layout_dict = fig.layout.to_plotly_json() if hasattr(fig, 'layout') else {}
            if 'width' in layout_dict or 'height' in layout_dict:
                layout_dict.pop('width', None)
                layout_dict.pop('height', None)
                fig.layout = go.Layout(**layout_dict)
        except Exception:
            pass
