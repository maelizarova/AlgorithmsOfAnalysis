import urllib.parse

if "?" in path:
        raw_path, query = path.split("?", 1)
        encoded_path = urllib.parse.quote(raw_path, safe="/")
        url = f"{BASE_URL}{encoded_path}?{query}"
    else:
        encoded_path = urllib.parse.quote(path, safe="/")
        url = f"{BASE_URL}{encoded_path}"
