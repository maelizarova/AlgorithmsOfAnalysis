def remove_columns_by_stoplist(df, stoplist, inplace=False, verbose=True):
    """
    Удаляет колонки из датафрейма, которые присутствуют в стоп-листе.
    
    Параметры:
    -----------
    df : pd.DataFrame
        Входной датафрейм
    stoplist : list или set
        Список колонок для удаления
    inplace : bool, optional (default=False)
        Изменять ли исходный датафрейм
    verbose : bool, optional (default=True)
        Выводить ли информацию об удаленных колонках
        
    Возвращает:
    -----------
    pd.DataFrame
        Дадафрейм с удаленными колонками (если inplace=False)
    """
    
    if not inplace:
        df = df.copy()
    
    # Преобразуем стоп-лист в set для быстрого поиска
    stoplist = set(stoplist)
    
    # Находим колонки, которые есть и в датафрейме, и в стоп-листе
    columns_to_drop = [col for col in df.columns if col in stoplist]
    
    if columns_to_drop:
        df.drop(columns=columns_to_drop, inplace=True)
        if verbose:
            print(f"Удалены колонки: {columns_to_drop}")
    elif verbose:
        print("Нет колонок для удаления (стоп-лист не совпадает с колонками датафрейма)")
    
    return df
