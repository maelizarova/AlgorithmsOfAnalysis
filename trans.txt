# %% [markdown]
# # Smoke-тесты для Admin API
# 
# Этот ноутбук позволяет:
# - поднять сервер один раз (если он еще не запущен),
# - запускать проверки API многократно без полного перезапуска,
# - запускать только нужные группы тестов (`cases`, `profiles`, `sessions`, `scores`).
# 
# Перед первым запуском:
# 1. Установите зависимости из `requirements.txt`.
# 2. Проверьте `config.yaml` (особенно `auth.api_key` и MinIO-параметры).
# 3. В ячейке ниже установите правильный `API_KEY`.

# %%
import json
import os
import subprocess
import sys
import time
import urllib.error
import urllib.request
from pathlib import Path

# Работает и когда ноутбук открыт из корня, и когда лежит в /notebooks
PROJECT_DIR = Path.cwd()
if not (PROJECT_DIR / "config.yaml").exists() and (PROJECT_DIR.parent / "config.yaml").exists():
    PROJECT_DIR = PROJECT_DIR.parent

APP_URL = "http://127.0.0.1:8000"
BASE_URL = f"{APP_URL}/api/admin"
API_KEY = "change-me"  # <-- поставьте ваш ключ из config.yaml (auth.api_key)
CONFIG_PATH = PROJECT_DIR / "config.yaml"

SERVER_PROCESS = None

print("PROJECT_DIR:", PROJECT_DIR)
print("BASE_URL:", BASE_URL)
print("CONFIG_PATH:", CONFIG_PATH)


# %%
def is_server_running(timeout: float = 1.5) -> bool:
    url = f"{APP_URL}/docs"
    try:
        with urllib.request.urlopen(url, timeout=timeout):
            return True
    except Exception:
        return False


def start_server_if_needed(wait_timeout: float = 20.0) -> None:
    """Запускает сервер, только если он еще не работает."""
    global SERVER_PROCESS

    if is_server_running():
        print("Сервер уже запущен. Повторный запуск не нужен.")
        return

    env = os.environ.copy()
    env["APP_CONFIG_PATH"] = str(CONFIG_PATH)

    cmd = [
        sys.executable,
        "-m",
        "uvicorn",
        "main:app",
        "--host",
        "127.0.0.1",
        "--port",
        "8000",
    ]

    SERVER_PROCESS = subprocess.Popen(
        cmd,
        cwd=str(PROJECT_DIR),
        env=env,
        stdout=subprocess.DEVNULL,
        stderr=subprocess.STDOUT,
    )

    start_ts = time.time()
    while time.time() - start_ts < wait_timeout:
        if is_server_running():
            print("Сервер успешно запущен.")
            return
        if SERVER_PROCESS.poll() is not None:
            raise RuntimeError("Uvicorn завершился раньше времени. Проверьте config.yaml и зависимости.")
        time.sleep(0.4)

    raise TimeoutError("Сервер не успел подняться за отведенное время.")


def stop_server_started_here() -> None:
    """Останавливает только тот сервер, который был запущен из этого ноутбука."""
    global SERVER_PROCESS

    if SERVER_PROCESS is None:
        print("В этом ноутбуке сервер не запускался — останавливать нечего.")
        return

    if SERVER_PROCESS.poll() is None:
        SERVER_PROCESS.terminate()
        try:
            SERVER_PROCESS.wait(timeout=5)
        except subprocess.TimeoutExpired:
            SERVER_PROCESS.kill()
            SERVER_PROCESS.wait(timeout=5)

    SERVER_PROCESS = None
    print("Сервер, запущенный из ноутбука, остановлен.")


# %%
def api_request(method: str, path: str, *, body: str | None = None, use_api_key: bool = True, timeout: float = 10.0):
    """Универсальный HTTP-запрос к API. Возвращает (status_code, payload)."""
    url = f"{BASE_URL}{path}"
    headers = {}
    data = None

    if use_api_key:
        headers["X-API-Key"] = API_KEY

    if body is not None:
        headers["Content-Type"] = "text/plain"
        data = body.encode("utf-8")

    request = urllib.request.Request(url=url, method=method.upper(), headers=headers, data=data)

    try:
        with urllib.request.urlopen(request, timeout=timeout) as response:
            raw = response.read().decode("utf-8")
            content_type = response.headers.get("Content-Type", "")
            if "application/json" in content_type and raw:
                return response.status, json.loads(raw)
            return response.status, raw
    except urllib.error.HTTPError as error:
        raw = error.read().decode("utf-8")
        try:
            payload = json.loads(raw) if raw else {}
        except json.JSONDecodeError:
            payload = raw
        return error.code, payload


def expect_status(actual: int, expected: int | tuple[int, ...], step: str) -> None:
    if isinstance(expected, tuple):
        if actual not in expected:
            raise AssertionError(f"{step}: ожидался один из {expected}, получен {actual}")
        return
    if actual != expected:
        raise AssertionError(f"{step}: ожидался {expected}, получен {actual}")


def print_result(step: str, status_code: int, payload) -> None:
    print(f"[{step}] status={status_code}")
    if isinstance(payload, (dict, list)):
        print(json.dumps(payload, ensure_ascii=False, indent=2))
    else:
        print(payload)
    print("-" * 80)


# %%
def test_auth_without_key() -> None:
    status_code, payload = api_request("GET", "/cases", use_api_key=False)
    print_result("auth_without_key", status_code, payload)
    expect_status(status_code, 401, "auth_without_key")


def test_cases_crud() -> None:
    yaml_v1 = "title: Smoke Case\nsteps:\n  - hello"
    yaml_v2 = "title: Smoke Case Updated\nsteps:\n  - hello\n  - updated"

    status_code, payload = api_request("POST", "/cases", body=yaml_v1)
    print_result("cases_create", status_code, payload)
    expect_status(status_code, 201, "cases_create")

    case_id = payload.get("id") if isinstance(payload, dict) else None
    if not case_id:
        raise AssertionError("cases_create: в ответе нет поля 'id'")

    status_code, payload = api_request("GET", "/cases")
    print_result("cases_list", status_code, payload)
    expect_status(status_code, 200, "cases_list")

    status_code, payload = api_request("GET", f"/cases/{case_id}")
    print_result("cases_get", status_code, payload)
    expect_status(status_code, 200, "cases_get")

    status_code, payload = api_request("PUT", f"/cases/{case_id}", body=yaml_v2)
    print_result("cases_update", status_code, payload)
    expect_status(status_code, 200, "cases_update")

    status_code, payload = api_request("DELETE", f"/cases/{case_id}")
    print_result("cases_delete", status_code, payload)
    expect_status(status_code, 200, "cases_delete")

    status_code, payload = api_request("GET", f"/cases/{case_id}")
    print_result("cases_get_after_delete", status_code, payload)
    expect_status(status_code, 404, "cases_get_after_delete")

    yaml_v1 = "name: Smoke Profile\nsegment: test"
    yaml_v2 = "name: Smoke Profile Updated\nsegment: test"

def test_sessions_list() -> None:
    status_code, payload = api_request("GET", "/sessions")
    print_result("sessions_list", status_code, payload)
    expect_status(status_code, 200, "sessions_list")


def test_score_get(score_id: str = "non-existent-score-id") -> None:
    status_code, payload = api_request("GET", f"/scores/{score_id}")
    print_result("score_get", status_code, payload)
    expect_status(status_code, (200, 404), "score_get")


# %%
def run_smoke(
    *,
    run_auth: bool = True,
    run_cases: bool = True,
    run_profiles: bool = True,
    run_sessions: bool = True,
    run_scores: bool = True,
    score_id: str = "non-existent-score-id",
    auto_start_server: bool = True,
):
    if auto_start_server:
        start_server_if_needed()

    checks = []

    if run_auth:
        test_auth_without_key()
        checks.append("auth")

    if run_cases:
        test_cases_crud()
        checks.append("cases")

    if run_profiles:
        test_profiles_crud()
        checks.append("profiles")

    if run_sessions:
        test_sessions_list()
        checks.append("sessions")

    if run_scores:
        test_score_get(score_id=score_id)
        checks.append("scores")

    print("Готово. Пройдены проверки:", ", ".join(checks) if checks else "(ничего не выбрано)")


# Базовый запуск всех проверок
run_smoke()
